import java.util.Scanner;

// Interface
interface Scholarship {
    void applyScholarship();
}

// Abstract Class
abstract class Student {

    String name;
    String usn;
    String course;
    double percentage;
    String college;

    abstract void displayDetails();
}

// Concrete Class
class MCAStudent extends Student implements Scholarship {

    @Override
    void displayDetails() {
        System.out.println("\n----- Student Details -----");
        System.out.println("Name       : " + name);
        System.out.println("USN        : " + usn);
        System.out.println("Course     : " + course);
        System.out.println("Percentage : " + percentage + "%");
        System.out.println("College    : " + college);
    }

    @Override
    public void applyScholarship() {
        if (percentage >= 75) {
            System.out.println("Scholarship Approved.");
        } else {
            System.out.println("Scholarship Not Approved.");
        }
    }
}

// Main Class
public class Main {
    public static void main(String[] args) {

        Scanner sc = new Scanner(System.in);
        MCAStudent student = new MCAStudent();

        System.out.print("Enter Name: ");
        student.name = sc.nextLine();

        System.out.print("Enter USN: ");
        student.usn = sc.nextLine();

        System.out.print("Enter Course: ");
        student.course = sc.nextLine();

        System.out.print("Enter Percentage: ");
        student.percentage = sc.nextDouble();
        sc.nextLine();

        System.out.print("Enter College: ");
        student.college = sc.nextLine();

        student.displayDetails();
        student.applyScholarship();

        sc.close();
    }
}

// SOLID Principles using Instagram Example

//---------------- SRP ----------------

// This class displays only user details
class InstagramUser {
    void displayUser() {
        String username = "Nandini";
        System.out.println("Username: " + username);
    }
}

//---------------- OCP & DIP ----------------

// New notification types can be added without modifying existing code
interface Notification {
    void send();
}

class LikeNotification implements Notification {
    public void send() {
        System.out.println("Like Notification Sent");
    }
}

class CommentNotification implements Notification {
    public void send() {
        System.out.println("Comment Notification Sent");
    }
}

//---------------- LSP ----------------

// Any child account can replace InstagramAccount
class InstagramAccount {
    void showAccountType() {
        System.out.println("Instagram Account");
    }
}

class PersonalAccount extends InstagramAccount {
    void showAccountType() {
        System.out.println("Personal Account");
    }
}

class BusinessAccount extends InstagramAccount {
    void showAccountType() {
        System.out.println("Business Account");
    }
}

//---------------- ISP ----------------

// Separate interfaces for separate features
interface PostFeature {
    void createPost();
}

interface ReelFeature {
    void uploadReel();
}

class Creator implements PostFeature, ReelFeature {
    public void createPost() {
        System.out.println("Creator created a post");
    }

    public void uploadReel() {
        System.out.println("Creator uploaded a reel");
    }
}

class Viewer implements PostFeature {
    public void createPost() {
        System.out.println("Viewer created a post");
    }
}

//---------------- DIP ----------------

// InstagramApp depends on abstraction (Notification)
class InstagramApp {
    Notification notification;

    InstagramApp(Notification notification) {
        this.notification = notification;
    }

    void notifyUser() {
        notification.send();
    }
}

// Main Class
public class SolidInstagramDemo {
    public static void main(String[] args) {

        // SRP
        InstagramUser user = new InstagramUser();
        user.displayUser();

        // OCP + DIP
        Notification n = new LikeNotification();
        InstagramApp app = new InstagramApp(n);
        app.notifyUser();

        // LSP
        InstagramAccount acc1 = new PersonalAccount();
        InstagramAccount acc2 = new BusinessAccount();

        acc1.showAccountType();
        acc2.showAccountType();

        // ISP
        Creator c = new Creator();
        c.createPost();
        c.uploadReel();

        Viewer v = new Viewer();
        v.createPost();
    }
}

package mca;

class ClassicSingleton {

    private static ClassicSingleton instance = null;

    private ClassicSingleton() {
        System.out.println("Singleton Object Created");
    }

    public static ClassicSingleton getInstance() {
        if (instance == null) {
            instance = new ClassicSingleton();
        }
        return instance;
    }
}

public class Sam {

    public static void main(String[] args) {

        ClassicSingleton obj1 = ClassicSingleton.getInstance();
        ClassicSingleton obj2 = ClassicSingleton.getInstance();

        System.out.println(obj1.hashCode());
        System.out.println(obj2.hashCode());

        if (obj1 == obj2) {
            System.out.println("Both references point to the same object");
        }
    }
}

package mca;

//Product Interface
interface Notification
{
    void send();
}

//Concrete Products
class EmailNotification implements Notification
{
    public void send()
    {
        System.out.println("Sending Email Notification");
    }
}

class SMSNotification implements Notification
{
    public void send()
    {
        System.out.println("Sending SMS Notification");
    }
}

class WhatsAppNotification implements Notification
{
    public void send()
    {
        System.out.println("Sending WhatsApp Notification");
    }
}

//Factory Class
class NotificationFactory
{
    public static Notification getNotification(String type)
    {
        if (type.equalsIgnoreCase("EMAIL"))
            return new EmailNotification();

        else if (type.equalsIgnoreCase("SMS"))
            return new SMSNotification();

        else if (type.equalsIgnoreCase("WHATSAPP"))
            return new WhatsAppNotification();

        return null;
    }
}

//Main Class
public class FactoryDemo
{
    public static void main(String[] args)
    {
        Notification notification = NotificationFactory.getNotification("Email");
        notification.send();
    }
}
